<script>

  const STORE_KEY = 'sok_bonuri_v1';

  const TOTAL_KEY = 'sok_total_v1';

  const MONTHS_RO = ['Ianuarie','Februarie','Martie','Aprilie','Mai','Iunie',

                     'Iulie','August','Septembrie','Octombrie','Noiembrie','Decembrie'];


  function loadReceipts() {

    try { return JSON.parse(localStorage.getItem(STORE_KEY)) || []; }

    catch { return []; }

  }

  function saveReceipts(list) { localStorage.setItem(STORE_KEY, JSON.stringify(list)); }


  let receipts = loadReceipts();

  let selected = new Set();


  const savedTotal = localStorage.getItem(TOTAL_KEY);

  if (savedTotal) document.getElementById('totalAmount').value = savedTotal;


  document.getElementById('totalAmount').addEventListener('input', e => {

    localStorage.setItem(TOTAL_KEY, e.target.value);

  });


  function formatDate(ts) {

    const d = new Date(ts);

    return `${d.getDate().toString().padStart(2,'0')}.${(d.getMonth()+1).toString().padStart(2,'0')}.${d.getFullYear()}`;

  }


  function showToast(msg) {

    const t = document.getElementById('toast');

    t.textContent = msg;

    t.classList.add('show');

    setTimeout(() => t.classList.remove('show'), 2500);

  }


  function rebuildFilters() {

    const months = new Set(), years = new Set();

    receipts.forEach(r => {

      const d = new Date(r.ts);

      months.add(d.getMonth()); years.add(d.getFullYear());

    });

    const mSel = document.getElementById('filterMonth');

    const ySel = document.getElementById('filterYear');

    const curM = mSel.value, curY = ySel.value;


    mSel.innerHTML = '<option value="">Toate lunile</option>';

    [...months].sort((a,b)=>a-b).forEach(m => {

      const o = document.createElement('option');

      o.value = m; o.textContent = MONTHS_RO[m];

      if (String(m) === curM) o.selected = true;

      mSel.appendChild(o);

    });


    ySel.innerHTML = '<option value="">Toți anii</option>';

    [...years].sort((a,b)=>b-a).forEach(y => {

      const o = document.createElement('option');

      o.value = y; o.textContent = y;

      if (String(y) === curY) o.selected = true;

      ySel.appendChild(o);

    });

  }


  function filteredReceipts() {

    const m = document.getElementById('filterMonth').value;

    const y = document.getElementById('filterYear').value;

    return receipts.filter(r => {

      const d = new Date(r.ts);

      if (m !== '' && d.getMonth() !== parseInt(m)) return false;

      if (y !== '' && d.getFullYear() !== parseInt(y)) return false;

      return true;

    });

  }


  function render() {

    const grid = document.getElementById('receiptsGrid');

    const list = filteredReceipts();

    const now = new Date();

    const thisMonth = receipts.filter(r => {

      const d = new Date(r.ts);

      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();

    }).length;


    document.getElementById('statTotal').textContent = receipts.length;

    document.getElementById('statMonth').textContent = thisMonth;

    document.getElementById('statSel').textContent = selected.size;


    const selBtn = document.getElementById('btnPrint');

    const selCount = document.getElementById('selCount');

    if (selected.size === 0) {

      selCount.innerHTML = 'Niciun bon selectat';

      selBtn.disabled = true;

    } else {

      selCount.innerHTML = `<strong>${selected.size}</strong> bon${selected.size > 1 ? 'uri' : ''} selectat${selected.size > 1 ? 'e' : ''}`;

      selBtn.disabled = false;

    }


    if (list.length === 0) {

      grid.innerHTML = `<div class="empty" style="grid-column:1/-1">

        <div class="empty-icon">🧾</div>

        <h3>${receipts.length > 0 ? 'NICIUN BON ÎN ACEASTĂ PERIOADĂ' : 'NICIUN BON ÎNCĂ'}</h3>

        <p>${receipts.length > 0 ? 'Schimbă filtrul de perioadă' : 'Apasă zona de sus pentru a adăuga primul bon'}</p>

      </div>`;

      rebuildFilters(); return;

    }


    const sorted = [...list].sort((a,b) => b.ts - a.ts);

    grid.innerHTML = sorted.map(r => `

      <div class="receipt-card ${selected.has(r.id) ? 'selected' : ''}" data-id="${r.id}">

        <div class="select-badge" data-sel="${r.id}">✓</div>

        <img src="${r.data}" alt="Bon" data-lb="${r.id}" loading="lazy">

        <div class="card-info">

          <div class="card-date">${formatDate(r.ts)}</div>

          <div class="card-name">${r.name || 'Bon ' + formatDate(r.ts)}</div>

  function addFiles(files) {

    let count = 0;

    const total = files.length;

    Array.from(files).forEach(file => {

      const reader = new FileReader();

      reader.onload = e => {

        receipts.push({ id: Date.now() + Math.random(), ts: Date.now(), name: file.name.replace(/\.[^.]+$/, ''), data: e.target.result });

        count++;

        if (count === total) { saveReceipts(receipts); render(); showToast(`✓ ${total} bon${total > 1 ? 'uri' : ''} adăugat${total > 1 ? 'e' : ''}`); }

      };

      reader.readAsDataURL(file);

    });

  }
